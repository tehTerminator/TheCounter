CREATE SCHEMA COUNTER;
CREATE TABLE COUNTER.PRODUCTS(ID INT NOT NULL AUTO_INCREMENT, TITLE CHAR(30), DESCRIPTION VARCHAR(255), RATE DOUBLE, IS_LIMITED BOOLEAN DEFAULT 'FALSE', PRIMARY KEY(ID) );
CREATE TABLE COUNTER.ACCOUNTS(ID INT NOT NULL AUTO_INCREMENT, TITLE CHAR(30), DESCRIPTION VARCHAR(255), BALANCE DOUBLE, PRIMARY KEY(ID) );
CREATE TABLE COUNTER.TRANSACTIONS(ID INT NOT NULL AUTO_INCREMENT, POSTED_ON DATETIME, FROM_ACCOUNT INT, TO_ACCOUNT INT, AMOUNT, PRIMARY_KEY(ID), FOREIGN KEY(FROM_ACCOUNT) REFERENCES ACCOUNTS.ID, FOREIGN KEY(TO_ACCOUNT) REFERENCES ACCOUNTS.ID);
CREATE TABLE COUNTER.ENTRY(POSTED_ON DATETIME, CLIENT CHAR(30), PRODUCT_ID INT, QUANTITY INT, AMOUNT DOUBLE);
CREATE TABLE COUNTER.STOCKS(PRODUCT_ID INT NOT NULL, QUANTITY INT, PRIMARY KEY(PRODUCT_ID), FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCTS.ID);
CREATE TABLE COUNTER.CUSTOMERS(ID INT NOT NULL AUTO_INCREMENT, NAME CHAR(50), COMPANY CHAR(100), ADDRESS VARCHAR(255), CONTACT CHAR(12), PRIMARY KEY(ID));
CREATE TABLE COUNTER.INVOICES(POSTED_ON DATETIME, CUSTOMER_ID INT, AMOUNT DOUBLE, PAYMENT_STATUS BOOLEAN, DESCRIPTION VARCHAR(255), PRIMARY KEY(POSTED_ON, CUSTOMER_ID) );
CREATE TABLE COUNTER.INVOICE_DATA(INVOICE_ID, PRODUCT_ID, QUANTITY, DISCOUNT, PRIMARY KEY(INVOICE_ID, PRODUCT_ID), FOREIGN KEY(INVOICE_ID) REFERENCES INVOICES.ID, FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCTS.ID);
CREATE TABLE COUNTER.VARIABLES(ID INT NOT NULL AUTO_INCREMENT, TITLE CHAR(50), DATA VARCHAR(255), PRIMARY KEY(ID));
